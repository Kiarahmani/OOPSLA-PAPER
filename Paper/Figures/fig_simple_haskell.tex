	data Effect = Eff String 
	type State = String 

	read :: State -> (String, Maybe Effect)
	read s = (s,Nothing)

	write :: String -> ((), Maybe Effect)
	write comment = ((), Just (Eff comment))

	apply :: State -> Effect -> State 
	apply s eff = let Eff comment = eff
		        	  in s ++ " - " ++ comment
	\end{lstlisting}
