\begin{theorem}
\label{theorem:two}
For all operation executions 
{\scriptsize $
(\E,op_{<s,i>}) 
    \xrightarrow{V}
  (\E',\eff) 
$},
the set of effects made visible to $\eta$ is maximal. i.e. for all
 $a \in V$:
\begin{fmathpar}
(a,\eta) \not\in \E'.\visZ \Rightarrow 
(\E'.A,\E'.\visZ \cup \{a,\eta\}, \E'.so) \not\models \psi[\eta/\hat{\eta}]
\end{fmathpar}
\end{theorem}
